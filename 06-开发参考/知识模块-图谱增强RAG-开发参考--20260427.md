### 3.1 节点类型 (Labels)

| 类型 | Label | 示例 |
|------|-------|------|
| 问题 | Question | 脚手架坠落事故 |
| 法规 | Law | 建设工程安全生产管理条例 |
| 条款 | Article | 第78条 |
| 案例 | Case | 2023年XX工地事故 |
| 政策 | Policy | JGJ59-2011 |
| 处罚 | Penalty | 罚款15万元 |
| 标准 | Standard | 建筑施工安全检查标准 |

### 3.2 关系类型 (Relationship Types)

| 关系 | Type | 方向 | 说明 |
|------|------|------|------|
| 涉及 | INVOLVES | -> | 问题涉及的法规 |
| 包含 | CONTAINS | -> | 法规包含条款 |
| 导致 | LEADS_TO | -> | 条款导致处罚 |
| 类似案例 | SIMILAR_CASE | -> | 类似案例关联 |
| 处罚依据 | BASIS | -> | 案例的处罚依据 |
| 支撑 | SUPPORTS | -> | 标准支撑条款 |
| 程序依据 | PROCEDURE_BASIS | -> | 处罚的程序依据 |

### 4.2 Neo4j 服务类

```python
# code/backend/app/services/neo4j_service.py

from neo4j import GraphDatabase
from typing import List, Dict, Optional

class Neo4jService:
    def __init__(self, uri: str = "bolt://172.20.36.91:7687",
                 user: str = "neo4j", password: str = "neo4j"):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))

    def close(self):
        self.driver.close()

    def query(self, cypher: str, params: Dict = None) -> List[Dict]:
        """执行 Cypher 查询"""
        with self.driver.session() as session:
            result = session.run(cypher, params or {})
            return [dict(record) for record in result]

    def get_node_by_name(self, name: str) -> Optional[Dict]:
        """根据名称查询节点"""
        cypher = "MATCH (n {name: $name}) RETURN n"
        results = self.query(cypher, {"name": name})
        return results[0]["n"] if results else None

    def get_node_relations(self, node_name: str) -> List[Dict]:
        """查询节点的所有关系"""
        cypher = """
        MATCH (n {name: $nodeName})-[r]-(connected)
        RETURN n, r, connected
        """
        return self.query(cypher, {"nodeName": node_name})

    def multi_hop_traverse(self, start_name: str, max_hops: int = 3) -> List[Dict]:
        """多跳遍历"""
        cypher = f"""
        MATCH path = (start {{name: $startName}})-[r*1..{max_hops}]->(end)
        WITH path, nodes(path) as ns, rels(path) as rs
        UNWIND ns as node
        UNWIND rs as rel
        RETURN DISTINCT node, rel
        """
        return self.query(cypher, {"startName": start_name})

    def get_reasoning_chain(self, question: str) -> List[Dict]:
        """获取推理链"""
        cypher = """
        MATCH path = (q:Question)-[:INVOLVES|SIMILAR_CASE*1..2]->(l:Law)
                          -[:CONTAINS]->(a:Article)
                          -[:LEADS_TO]->(p:Penalty)
        WHERE q.name = $question
        RETURN path
        """
        return self.query(cypher, {"question": question})

    def create_entity(self, label: str, properties: Dict) -> Dict:
        """创建实体"""
        props_str = ", ".join([f"{k}: ${k}" for k in properties.keys()])
        cypher = f"CREATE (n:{label} {{{props_str}}}) RETURN n"
        results = self.query(cypher, properties)
        return results[0]["n"] if results else None

    def create_relation(self, source_name: str, target_name: str,
                       rel_type: str, properties: Dict = None) -> Dict:
        """创建关系"""
        cypher = f"""
        MATCH (a {{name: $sourceName}})
        MATCH (b {{name: $targetName}})
        CREATE (a)-[r:{rel_type}]->(b)
        RETURN r
        """
        params = {"sourceName": source_name, "targetName": target_name}
        if properties:
            for k, v in properties.items():
                params[k] = v
        results = self.query(cypher, params)
        return results[0]["r"] if results else None

    def delete_all(self):
        """删除所有数据（谨慎使用）"""
        cypher = "MATCH (n) DETACH DELETE n"
        self.query(cypher)


# 全局实例
neo4j_service = Neo4jService()
```

## 五、Cypher 查询示例

### 5.1 创建示例图谱

```cypher
// 创建问题节点
CREATE (q1:Question {
  id: 'q1',
  name: '脚手架坠落事故',
  description: '工地脚手架搭建不符合规范导致坠落事故'
})

// 创建法规节点
CREATE (l1:Law {
  id: 'n1',
  name: '建设工程安全生产管理条例',
  description: '国务院发布的安全生产管理条例'
})

// 创建条款节点
CREATE (a1:Article {
  id: 'n2',
  name: '第78条',
  description: '未按规定设置安全防护设施的处罚规定'
})

// 创建案例节点
CREATE (c1:Case {
  id: 'n3',
  name: '2023年XX工地事故',
  description: '某工地脚手架坠落致2人死亡事故'
})

// 创建处罚节点
CREATE (p1:Penalty {
  id: 'n6',
  name: '罚款15万元',
  description: '依据安全生产管理条例第78条的处罚决定'
})

// 创建关系
CREATE (q1)-[:INVOLVES]->(l1)
CREATE (l1)-[:CONTAINS]->(a1)
CREATE (a1)-[:LEADS_TO]->(p1)
CREATE (q1)-[:SIMILAR_CASE]->(c1)
CREATE (c1)-[:BASIS]->(l1)
```

### 5.2 查询某个节点的关联

```cypher
MATCH (n {name: '脚手架坠落事故'})-[r]-(connected)
RETURN n, r, connected
```

### 5.3 多跳查询

```cypher
// 查询从问题到处罚的完整路径
MATCH path = (q:Question)-[:INVOLVES]->(l:Law)-[:CONTAINS]->(a:Article)-[:LEADS_TO]->(p:Penalty)
WHERE q.name = '脚手架坠落事故'
RETURN path
```

### 5.4 查询所有实体

```cypher
MATCH (n)
WHERE labels(n)[0] IN ['Question', 'Law', 'Article', 'Case', 'Policy', 'Penalty', 'Standard']
RETURN n
```

## 六、后端 Graph API（待开发）

```python
# app/api/v1/graph.py

from fastapi import APIRouter, Depends
from pydantic import BaseModel
from typing import List, Optional
from app.services.neo4j_service import neo4j_service

router = APIRouter(prefix="/graph", tags=["graph"])

class EntityCreate(BaseModel):
    label: str
    name: str
    description: Optional[str] = None
    properties: Optional[dict] = None

class RelationCreate(BaseModel):
    source_name: str
    target_name: str
    relation_type: str

@router.get("/search")
async def search_graph(question: str, max_hops: int = 3):
    """图谱搜索"""
    # 1. 查询推理链
    chain = neo4j_service.get_reasoning_chain(question)

    # 2. 多跳遍历获取图数据
    graph_data = neo4j_service.multi_hop_traverse(question, max_hops)

    return {
        "reasoning_chain": chain,
        "graph_data": graph_data
    }

@router.get("/entities")
async def list_entities():
    """获取所有实体"""
    cypher = """
    MATCH (n)
    WHERE labels(n)[0] IN ['Question', 'Law', 'Article', 'Case', 'Policy', 'Penalty', 'Standard']
    RETURN n
    """
    return neo4j_service.query(cypher)

@router.post("/entity")
async def create_entity(entity: EntityCreate):
    """创建实体"""
    properties = {
        "id": str(uuid.uuid4()),
        "name": entity.name,
        "description": entity.description or ""
    }
    if entity.properties:
        properties.update(entity.properties)
    return neo4j_service.create_entity(entity.label, properties)

@router.post("/relation")
async def create_relation(relation: RelationCreate):
    """创建关系"""
    return neo4j_service.create_relation(
        relation.source_name,
        relation.target_name,
        relation.relation_type
    )
```

## 九、常见问题

### Q1: Neo4j 连接失败？

A: 检查容器是否运行 `docker ps | grep neo4j`，检查端口是否映射 `docker port neo4j`

### Q2: 如何重置图谱数据？

```cypher
MATCH (n) DETACH DELETE n
```

### Q3: 如何查看所有节点和关系？

```cypher
MATCH (n) RETURN n
MATCH ()-[r]->() RETURN r
```
