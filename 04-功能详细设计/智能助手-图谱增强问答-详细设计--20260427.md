### 2. Neo4j Python 客户端

```python
# 使用 py2neo 或 neo4j-driver
from neo4j import GraphDatabase

class Neo4jService:
    def __init__(self, uri, user, password):
        self.driver = GraphDatabase.driver(uri, auth=(user, password))

    def query(self, cypher, params=None):
        with self.driver.session() as session:
            result = session.run(cypher, params)
            return [dict(record) for record in result]

    def close(self):
        self.driver.close()
```

### 3. 核心 Cypher 查询

#### 3.1 查询节点的所有关系

```python
def get_node_relations(node_name: str):
    cypher = """
    MATCH (n {name: $nodeName})-[r]-(connected)
    RETURN n, r, connected
    """
    return neo4j.query(cypher, {"nodeName": node_name})
```

#### 3.2 多跳遍历

```python
def multi_hop_traverse(start_name: str, max_hops: int = 3):
    cypher = f"""
    MATCH path = (start {{name: $startName}})-[r*1..{max_hops}]->(end)
    WITH path, nodes(path) as ns, rels(path) as rs
    UNWIND ns as node
    UNWIND rs as rel
    RETURN DISTINCT node, rel
    """
    return neo4j.query(cypher, {"startName": start_name})
```

#### 3.3 推理路径查询

```python
def get_reasoning_path(question: str):
    cypher = """
    MATCH path = (q:Question)-[:INVOLVES|SIMILAR_CASE*1..2]->(l:Law)
                      -[:CONTAINS]->(a:Article)
                      -[:LEADS_TO]->(p:Penalty)
    WHERE q.name = $question
    RETURN path
    """
    return neo4j.query(cypher, {"question": question})
```

#### 3.4 创建节点和关系

```python
def create_entity_and_relations(entity: dict, relations: list):
    # 创建实体
    cypher = f"""
    CREATE (n:{entity['type']} {{id: $id, name: $name, description: $description}})
    """
    neo4j.query(cypher, entity)

    # 创建关系
    for rel in relations:
        cypher = f"""
        MATCH (a {{name: $sourceName}})
        MATCH (b {{name: $targetName}})
        CREATE (a)-[r:{rel['type']}]->(b)
        """
        neo4j.query(cypher, rel)
```
